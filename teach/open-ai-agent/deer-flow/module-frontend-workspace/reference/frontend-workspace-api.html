Frontend workspace 微观 API 参考

线程创建易错：isNewThread 表示后端是否已有线程，控制 SDK 是否拉 /history；isWelcomeMode 仅为欢迎布局视觉态。新线程创建成功后必须用 history.replaceState 更新 URL，不要用 Next.js router.push，否则会 remount 丢状态（issue #2746）。

URL 查询参数

参数适用路由含义
?mock=true 聊天线程页 useThreadChat().isMock；禁用真实 API 提交
?mode=skill /workspace/chats/new 触发 useSpecificChatMode 预填技能创建提示
?thread_id= /workspace/scheduled-tasks 过滤某线程的定时任务；默认 context_mode 为 reuse_thread

调用示例

新聊天页组合（简化）

// chats/[thread_id]/page.tsx 核心结构
const { threadId, isNewThread, setThreadId, setIsNewThread, isMock } = useThreadChat();
const { thread, sendMessage, ... } = useThreadStream({
  threadId: isNewThread ? undefined : threadId,
  displayThreadId: threadId,
  onStart: (id) => {
    history.replaceState(null, "", `/workspace/chats/${id}`);
    setThreadId(id);
    setIsNewThread(false);
  },
});

return (
  <ThreadContext.Provider value={{ thread, isMock }}>
    <SidecarProvider parentThreadId={threadId} context={settings.context}>
      <ChatBox threadId={threadId}>
        <MessageList threadId={threadId} thread={thread} ... />
        <InputBox threadId={threadId} onSubmit={handleSubmit} ... />
      </ChatBox>
    </SidecarProvider>
  </ThreadContext.Provider>
);

删线程后重置本地聊天态

import { resetThreadChatAfterDelete } from "@/components/workspace/chats";

resetThreadChatAfterDelete({
  deletedThreadId: thread.thread_id,
  nextPath: "/workspace/chats/new",
});
router.replace("/workspace/chats/new");
