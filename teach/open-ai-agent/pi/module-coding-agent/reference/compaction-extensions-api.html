export function calculateContextTokens(usage: Usage): number

边界条件无 assistant 消息时 usageTokens=0，lastUsageIndex=null

export estimateTokens

export function estimateTokens(message: AgentMessage): number
参数message: AgentMessage — 单条消息
返回值number — 估算 token 数（使用 chars/4 启发式，保守估算）
行为按消息角色 (user/assistant/custom/bashExecution/branchSummary/compactionSummary) 计算字符数并向上取整除 4
边界条件unknow 角色返回 0
注意：图片内容估算为 4800 字符 (约 1200 tokens)，该值为保守估算。

export findCutPoint

export function findCutPoint( entries: SessionEntry[], startIndex: number, endIndex: number, keepRecentTokens: number, ): CutPointResult
参数entries: 会话条目; startIndex: 起始索引（含）; endIndex: 结束索引（不含）; keepRecentTokens: 保留 token 预算
返回值CutPointResult — { firstKeptEntryIndex, turnStartIndex, isSplitTurn }
算法从最新消息倒序累加估算 token，超出预算时在最近合法切割点（user/assistant/bashExecution/custom/branchSummary 消息）切割
切割规则绝不切割 toolResult（必须跟随其 tool call）；若切割点不是 user 消息，标记为 split turn 并向前查找 turn 起点
关键约束：切割点必须在 user 或 assistant 消息上，不能切在 toolResult 上。当从 assistant 消息切割时，其后续 toolResult 会被一并保留。

// 创建扩展对象（模拟加载产物）
function createExtension(
  onBeforeCompact?: (event: SessionBeforeCompactEvent) => { cancel?: boolean; compaction?: any } | undefined,
  onCompact?: (event: SessionCompactEvent) => void,
): Extension {
  const handlers = new Map<string, ((event: any, ctx: any) => Promise<any>)[]>();

  handlers.set("session_before_compact", [
    async (event: SessionBeforeCompactEvent) => {
      capturedEvents.push(event);
      if (onBeforeCompact) return onBeforeCompact(event);
      return undefined;
    },
  ]);

  handlers.set("session_compact", [
    async (event: SessionCompactEvent) => {
      capturedEvents.push(event);
      if (onCompact) onCompact(event);
      return undefined;
    },
  ]);

  return {
    path: "test-extension",
    resolvedPath: "/test/test-extension.ts",
    sourceInfo: createSyntheticSourceInfo("<test:test-extension>", { source: "test" }),
    handlers,
    tools: new Map(),
    messageRenderers: new Map(),
    commands: new Map(),
    flags: new Map(),
    shortcuts: new Map(),
  };
}

// 监听压缩事件
const extension = createExtension(
  (event) => {
    console.log("即将压缩，消息数:", event.preparation.messagesToSummarize.length);
    // 返回 { cancel: true } 可取消压缩
    // 返回 { compaction: myResult } 可替换压缩结果
    return undefined; // 不干预
  },
  (event) => {
    console.log("压缩完成:", event.compactionEntry.summary);
    console.log("触发原因:", event.reason); // "manual" | "threshold" | "overflow"
  },
);

3.2 手动触发上下文压缩

import { compact, prepareCompaction, DEFAULT_COMPACTION_SETTINGS } from "@earendil-works/pi-coding-agent";

async function manualCompact(sessionEntries, model, apiKey) {
  const preparation = prepareCompaction(sessionEntries, DEFAULT_COMPACTION_SETTINGS);

  if (!preparation) {
    console.log("无需压缩");
    return;
  }

  console.log(`待摘要消息: ${preparation.messagesToSummarize.length} 条`);
  console.log(`当前 tokens: ${preparation.tokensBefore}`);
  console.log(`文件操作: read=${preparation.fileOps.read.size}, modified=${preparation.fileOps.edited.size}`);

  const result = await compact(preparation, model, apiKey, undefined, "关注 API 变更");

  console.log("摘要:", result.summary);
  console.log("保留首条 ID:", result.firstKeptEntryId);
  console.log("读取文件:", result.details?.readFiles);
  console.log("修改文件:", result.details?.modifiedFiles);
}

import { isBashToolResult, isEditToolResult, isReadToolResult } from "@earendil-works/pi-coding-agent";

// 在 tool_result 事件处理器中
pi.on("tool_result", (event, ctx) => {

    console.log("Bash 退出码:", event.details?.exitCode);
    // 修改错误标志

  // 不修改其他工具的结果
  return undefined;
});
