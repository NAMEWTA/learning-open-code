2. 四个核心工具接口对照

工具ToolNameTypeBox SchemaOperations 接口创建函数

命令 "bash" {command, timeout?} BashOperations {exec} createBashTool / createBashToolDefinition
写入 "write" {path, content} WriteOperations {writeFile, mkdir} createWriteTool / createWriteToolDefinition
编辑 "edit" {path, edits[]} EditOperations {readFile, writeFile, access} createEditTool / createEditToolDefinition

3. ToolDefinition 接口（扩展系统类型）

// extensions/types.ts — ToolDefinition 的完整接口（概念，具体以源码为准）
interface ToolDefinition<TInput, TDetails, TRenderState = unknown> {
  name: string;                      // 工具名
  label?: string;                    // UI 标签
  description: string;               // LLM 的函数描述
  promptSnippet?: string;            // 系统提示中的简短描述
  promptGuidelines?: string[];       // 使用指南（注入系统提示）
  parameters: TSchema;               // TypeBox schema
  executionMode?: "serial" | "parallel"; // 执行模式
  renderShell?: "self" | "default";  // 渲染策略（edit 用 "self"）
  prepareArguments?: (input: unknown) => TInput;  // 参数预处理
  execute(
    toolCallId: string,
    params: TInput,
    signal?: AbortSignal,
    onUpdate?: (partial: PartialResult) => void,
    ctx?: ExtensionContext,
  ): Promise<{content: ContentBlock[]; details: TDetails | undefined}>;
  renderCall?(args: unknown, theme, context): Component;   // 调用展示
  renderResult?(result, options, theme, context): Component; // 结果展示
}

4. AgentTool 接口（Agent 运行时类型）

// @earendil-works/pi-agent-core — AgentTool 接口
interface AgentTool<TInput, TDetails = unknown> {
  name: string;
  label?: string;
  description: string;
  parameters: TSchema;
  preparedToolCall?: boolean;
  executionMode?: "serial" | "parallel";
  execute(
    toolCallId: string,
    params: TInput,
    signal?: AbortSignal,
    onUpdate?: (partial: PartialResult) => void,
  ): Promise<{content: ContentBlock[]; details?: TDetails}>;
}

5. 工具分组策略

分组工具列表用途
createCodingToolDefinitionsread, bash, edit, write默认激活，支持读写操作
createReadOnlyToolDefinitionsread, grep, find, ls只读模式，禁止修改文件系统
createAllToolDefinitions全部 7 个工具完整注册表，供过滤使用
