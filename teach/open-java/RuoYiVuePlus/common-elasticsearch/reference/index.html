定位

对 Easy-Es（Dromara 社区出品的 Elasticsearch ORM 框架，3.0.2 版本）的项目化适配层。模块仅 2 个 Java 类，通过 @EsMapperScan 自动发现 Mapper、通过 @ConditionalOnProperty 实现可插拔、通过 EnvironmentPostProcessor 同步健康检查开关。API 设计对标 MyBatis-Plus，降低 ES 学习成本。

定义 ES 文档实体

@Data
@IndexName("document")      // ES 索引名称
public class Document {
    @IndexId                // 标记文档 ID
    private String id;

    @IndexField("title")    // 映射 ES 字段
    private String title;

    @IndexField("content")
    private String content;
}

CRUD 查询示例

@RestController
public class EsCrudController {
    @Autowired
    private DocumentMapper documentMapper;

    // 精确查询
    public List<Document> search(String title) {
        LambdaEsQueryWrapper<Document> wrapper = new LambdaEsQueryWrapper<>();
        wrapper.eq(Document::getTitle, title);
        return documentMapper.selectList(wrapper);
    }

    // 模糊搜索
    public List<Document> fuzzySearch(String keyword) {
        LambdaEsQueryWrapper<Document> wrapper = new LambdaEsQueryWrapper<>();
        wrapper.like(Document::getContent, keyword);
        return documentMapper.selectList(wrapper);
    }

    // 分页查询
    public PageResult<Document> page(int page, int size) {
        IPage<Document> esPage = new Page<>(page, size);
        return documentMapper.page(esPage);
    }
}

EnvironmentPostProcessor — 健康检查开关同步

@Override
public void postProcessEnvironment(ConfigurableEnvironment env,
                                    SpringApplication application) {
    String esEnable = env.getProperty("easy-es.enable");
    if ("false".equals(esEnable)) {
        // 关闭 Easy-Es 时，同步关闭 ES 健康检查，避免启动报错
        Map<String, Object> props = new HashMap<>();
        props.put("management.health.elasticsearch.enabled", "false");
        env.getPropertySources()
           .addLast(new MapPropertySource("actuator-elasticsearch", props));
    }
}

@Override
public int getOrder() {
    return Ordered.HIGHEST_PRECEDENCE;  // 确保最先执行
}
