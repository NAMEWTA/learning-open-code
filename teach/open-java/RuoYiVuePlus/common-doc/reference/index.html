JavadocOperationCustomizer 核心流程

@Override
public Operation customize(Operation operation, HandlerMethod handlerMethod) {
    Javadoc javadoc = javadocProvider.getClassJavadoc(handlerMethod.getBeanType());
    String methodJavadoc = javadocProvider
        .getMethodJavadocDescription(handlerMethod.getMethod());

    // 1. JavaDoc 首句 → summary
    operation.setSummary(javadocProvider.getFirstSentence(methodJavadoc));
    // 2. JavaDoc 全文 → description
    operation.setDescription(methodJavadoc);

    // 3. 遍历 JavadocResolver 扩展链，追加权限描述等
    for (JavadocResolver resolver : resolvers) {
        if (resolver.supports(handlerMethod)) {
            resolver.resolve(handlerMethod, operation);
        }
    }
    return operation;
}

PlusPaths 标记类 — 防止路径重复

// SpringDocConfig 内部
static class PlusPaths extends Paths { }  // 无新增逻辑，纯标记类

// openApiCustomizer 中使用
GlobalOpenApiCustomizer customizer = openAPI -> {
    Paths oldPaths = openAPI.getPaths();
    if (oldPaths instanceof PlusPaths) {
        return;  // 已处理过，跳过避免重复添加前缀
    }
    // ... 给每个 path 添加 context-path 前缀 ...
    openAPI.setPaths(new PlusPaths());  // 标记已处理
};
