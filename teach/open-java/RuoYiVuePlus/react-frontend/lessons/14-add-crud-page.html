步骤 2：定义 API 函数

src/api/system/notice/index.ts（新建）

import { service } from '@/api/request';
import type { R, PageResult } from '@/api/types';
import type { Notice, NoticeQuery, NoticeFormData } from './types';

// 分页列表
export const getNoticeList = (params: NoticeQuery) =>
  service.get<R<PageResult<Notice>>>('/system/notice/list', { params });

// 详情
export const getNotice = (noticeId: number) =>
  service.get<R<Notice>>(`/system/notice/${noticeId}`);

// 新增
export const addNotice = (data: NoticeFormData) =>
  service.post<R<void>>('/system/notice', data);

// 修改
export const updateNotice = (noticeId: number, data: NoticeFormData) =>
  service.put<R<void>>(`/system/notice/${noticeId}`, data);

// 删除
export const deleteNotice = (ids: number[]) =>
  service.delete<R<void>>(`/system/notice/${ids.join(',')}`);

步骤 3：编写页面组件

src/pages/system/notice/index.tsx（新建）

import { PageContainer, ProTable, ModalForm, ProFormText, ProFormSelect } from '@ant-design/pro-components';
import type { ProColumns, ActionType } from '@ant-design/pro-components';
import { Button, message, Popconfirm, Tag } from 'antd';
import { PlusOutlined } from '@ant-design/icons';
import { useRef, useState } from 'react';
import { useDict } from '@/hooks/useDict';
import type { Notice, NoticeFormData } from '@/api/system/notice/types';
import { getNoticeList, addNotice, updateNotice, deleteNotice } from '@/api/system/notice';

const NoticePage: React.FC = () => {
  // ===== 状态 =====
  const actionRef = useRef<ActionType>();           // ProTable ref（调 reload 用）
  const [modalOpen, setModalOpen] = useState(false);
  const [editingRecord, setEditingRecord] = useState<Notice | undefined>();
  const isEdit = !!editingRecord;

  // ===== 字典 =====
  const { data: noticeTypeDict } = useDict('sys_notice_type');
  const { data: statusDict } = useDict('sys_normal_disable');

  // ===== 列定义 =====
  const columns: ProColumns<Notice>[] = [
    { title: '公告ID', dataIndex: 'noticeId', hideInSearch: true, width: 80 },
    {
      title: '公告标题',
      dataIndex: 'noticeTitle',
      valueType: 'text',            // 搜索：Input
    },
    {
      title: '公告类型',
      dataIndex: 'noticeType',
      valueType: 'select',          // 搜索：Select
      valueEnum: noticeTypeDict?.reduce((acc, d) => {
        acc[d.value] = { text: d.label };
        return acc;
      }, {} as any),
      render: (_, r) => {
        const item = noticeTypeDict?.find(d => d.value === r.noticeType);
        return <Tag color={r.noticeType === '1' ? 'blue' : 'green'}>{item?.label}</Tag>;
      },
    },
    {
      title: '状态',
      dataIndex: 'status',
      valueType: 'select',
      valueEnum: statusDict?.reduce((acc, d) => {
        acc[d.value] = { text: d.label };
        return acc;
      }, {} as any),
      render: (_, r) => {
        const item = statusDict?.find(d => d.value === r.status);
        return <Tag color={r.status === '0' ? 'success' : 'error'}>{item?.label}</Tag>;
      },
    },
    { title: '创建者', dataIndex: 'createBy', hideInSearch: true },
    { title: '创建时间', dataIndex: 'createTime', valueType: 'dateRange', hideInSearch: false },
    {
      title: '操作',
      valueType: 'option',
      render: (_, record) => [
        <Button key="edit" type="link" onClick={() => {
          setEditingRecord(record);
          setModalOpen(true);
        }}>编辑</Button>,
        <Popconfirm key="del" title="确认删除？" onConfirm={async () => {
          await deleteNotice([record.noticeId]);
          message.success('删除成功');
          actionRef.current?.reload();
        }}><Button type="link" danger>删除</Button></Popconfirm>,
      ],
    },
  ];

  // ===== 工具栏 =====
  const toolBarRender = () => [
    <Button type="primary" icon={<PlusOutlined />} onClick={() => {
      setEditingRecord(undefined);  // 清空 = 新增模式
      setModalOpen(true);
    }}>新增公告</Button>,
  ];

  // ===== 渲染 =====
  return (
    <PageContainer>
      <ProTable<Notice>
        headerTitle="公告列表"
        actionRef={actionRef}
        rowKey="noticeId"
        columns={columns}
        request={async (params) => {
          const res = await getNoticeList({
            pageNum: params.current!,
            pageSize: params.pageSize!,
            ...params,
          });
          return { data: res.data.rows, total: res.data.total, success: true };
        }}
        toolBarRender={toolBarRender}
      />

      {/* 新增/编辑弹窗 */}
      <ModalForm<NoticeFormData>
        title={isEdit ? '编辑公告' : '新增公告'}
        open={modalOpen}
        onOpenChange={setModalOpen}
        initialValues={editingRecord}
        onFinish={async (values) => {
          if (isEdit) {
            await updateNotice(editingRecord.noticeId, values);
          } else {
            await addNotice(values);
          }
          message.success(isEdit ? '修改成功' : '新增成功');
          actionRef.current?.reload();
          return true;
        }}
      >
        <ProFormText name="noticeTitle" label="公告标题" rules={[{ required: true }]} />
        <ProFormSelect name="noticeType" label="公告类型" valueEnum={{
          '1': '通知', '2': '公告',
        }} rules={[{ required: true }]} />
        <ProFormText name="noticeContent" label="内容" />
        <ProFormSelect name="status" label="状态" valueEnum={{
          '0': '正常', '1': '关闭',
        }} />
      </ModalForm>
    </PageContainer>
  );
};

export default NoticePage;

复盘：你添加这个页面涉及了多少知识点？

✅ TypeScript 类型定义（第 1-2 节） → Notice, NoticeQuery, NoticeFormData 三个 interface ✅ React 组件结构（第 2 节） → 函数组件 + Props（本页无外部 props）+ JSX 渲染 ✅ useState + useRef（第 3-4 节） → modalOpen, editingRecord 控制弹窗状态 → actionRef 引用 ProTable 实例 ✅ 自定义 Hook useDict（第 4 节） → 字典数据获取 + 缓存 ✅ Zustand Store（第 7-8 节） → ProTable request 内部通过 request.ts 自动附加 token ✅ Axios 拦截器链（第 9 节） → addNotice → service.post → 请求拦截器（+token） → 发送 ✅ 持久化（第 10 节） → Token 来自 localStorage，自动携带 ✅ 动态路由（第 11-12 节） → DynamicPage 中的 migratedPages 映射 ✅ ProTable CRUD 模式（第 13 节） → columns + request + toolBarRender + ModalForm
