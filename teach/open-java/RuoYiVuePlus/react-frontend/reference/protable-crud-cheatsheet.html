ProTable request 函数签名

const fetchList = async (params: {
  current?: number;    // ProTable 传入
  pageSize?: number;   // ProTable 传入
  // ... 所有搜索字段自动合并!
}) => {
  const res = await api.getList({
    pageNum: params.current!,
    pageSize: params.pageSize!,
    ...params,
  });
  return {
    data: res.data.rows,
    total: res.data.total,
    success: true,
  };
};

Column 常用 valueType

valueType搜索控件表格显示
textInput纯文本
selectSelect（需配 valueEnum）按 valueEnum 映射
dateRangeDatePicker.RangePicker日期范围文字
dateDatePicker日期文字

render 函数模式

// 模式 1：简单标签
render: (_, record) => <Tag>{record.status}</Tag>

// 模式 2：字典映射
render: (_, record) => {
  const item = dict?.find(d => d.value === record.status);
  return <Tag color={item?.color}>{item?.label}</Tag>;
}

// 模式 3：交互组件
render: (_, record) => (
  <Switch checked={record.status === '0'} onChange={...} />
)

// 模式 4：操作按钮组
render: (_, record) => [
  <Button key="edit">编辑</Button>,
  <Button key="del" danger>删除</Button>,
]

新增/编辑共用弹窗模式

const [open, setOpen] = useState(false);
const [record, setRecord] = useState<T>();
const isEdit = !!record;

// 新增按钮
onClick={() => { setRecord(undefined); setOpen(true); }}

// 编辑按钮
onClick={() => { setRecord(row); setOpen(true); }}

// ModalForm
<ModalForm
  title={isEdit ? '编辑' : '新增'}
  open={open}
  initialValues={record}        // 编辑时回填
  onOpenChange={setOpen}
  onFinish={async (values) => {
    if (isEdit) await updateApi(record.id, values);
    else await addApi(values);
    actionRef.current?.reload(); // 刷新表格
    return true;                 // 关闭弹窗
  }}
>
