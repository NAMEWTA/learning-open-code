核心 Composable 详解

useTableSelection — 表格多选

const { ids, single, multiple, handleSelectionChange } =
  useTableSelection<UserVO>(row => row.userId);
// single=true  → 选中数!=1 时禁用单个操作按钮
// multiple=true → 选中数=0 时禁用批量操作按钮
// 语义：直接绑定 el-button :disabled="single"

useSearchReset — 搜索重置

const { resetQuery } = useSearchReset({
  queryFormRef, queryParams,
  pageNumKey: 'pageNum', pageSizeKey: 'pageSize',
  initialPageNum: 1, initialPageSize: 10,
  afterReset: getList  // 重置后自动刷新列表
});

useFormDialog — 表单弹窗

const { dialog, openDialog, closeDialog } = useFormDialog({
  form: ref<UserForm>(...),
  formRef: ref<ElFormInstance>(),
  initialFormData: { userName: '', status: '0', ... },
  initialTitle: '新增用户'
});
// openDialog('新增用户') → 先 resetForm() 再打开
// showDialog('编辑用户') → 不重置，保留已填充数据

useLoading — 加载态高阶包装

const { loading, withLoading } = useLoading();
const getList = async () => {
  await withLoading(() => listUser(queryParams));
  // loading 自动 true → false，finally 保证关闭
};

典型 CRUD 页面组合模式

// 一个标准 CRUD 页面同时使用 4-5 个 composables 1. useTableSelection → 表格多选 → ids, single, multiple 2. useSearchToggle → 搜索展开 → showSearch 3. useSearchReset → 搜索重置 → resetQuery(afterReset: getList) 4. useDateRangeQuery → 日期范围 → applyDateRange(params) 5. useFormDialog → 新增/编辑弹窗 → openDialog 6. useLoading → 加载态 → withLoading(getList) 组合后只需在 getList 中串联调用即可
设计优势：Vue composable 可在条件分支中调用，无 React Hooks 的调用顺序约束。ref/computed 自动追踪依赖，无需显式依赖数组。
