API 函数模板

import type { UserForm, UserQuery, UserVO } from './types';
import type { AxiosPromise } from '@/utils/api-types';
import request from '@/utils/request';

/** 查询用户列表 */
export const listUser = (query: UserQuery): AxiosPromise<PageResult<UserVO>> => {
  return request({ url: '/system/user/list', method: 'get', params: query });
};

/** 新增用户 */
export const addUser = (data: UserForm) => {
  return request({ url: '/system/user', method: 'post', data });
};

/** 敏感操作（需加密+防重复） */
export const resetUserPwd = (userId, password) => {
  return request({
    url: '/system/user/resetPwd', method: 'put',
    headers: { isEncrypt: true, repeatSubmit: false },
    data: { userId, password }
  });
};

// 全局类型（src/api/types.ts）

// AxiosPromise 包装（src/utils/api-types.ts）
type AxiosPromise<T> = Promise<RuoYiAjaxResult<T>>
interface RuoYiAjaxResult<T> { code: number; msg: string; data: T }
