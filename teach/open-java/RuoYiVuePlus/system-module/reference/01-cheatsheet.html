6. 新建 Service 实现

@RequiredArgsConstructor
@Service
public class SysXxxServiceImpl implements ISysXxxService {
    private final SysXxxMapper xxxMapper;

    @Override
    public PageResult<SysXxxVo> selectPageXxxList(SysXxxBo bo, PageQuery pageQuery) {
        Page<SysXxxVo> page = xxxMapper.selectPageXxxList(pageQuery.build(), bo);
        return PageResult.build(page.getRecords(), page.getTotal());
    }

    @Override
    @Transactional(rollbackFor = Exception.class)
    public int insertXxx(SysXxxBo bo) {
        SysXxx entity = MapstructUtils.convert(bo, SysXxx.class);
        return xxxMapper.insert(entity);
    }

    @Override
    public boolean checkXxxNameUnique(SysXxxBo bo) {
        boolean exist = xxxMapper.lambda()
            .eq(SysXxx::getXxxName, bo.getXxxName())
            .neIfPresent(SysXxx::getXxxId, bo.getXxxId())
            .exists();
        return !exist;
    }
}

7. 新建 Controller

@Validated
@RequiredArgsConstructor
@RestController
@RequestMapping("/system/xxx")
public class SysXxxController extends BaseController {
    private final ISysXxxService xxxService;

    @SaCheckPermission("system:xxx:list")
    @GetMapping("/list")
    public R<PageResult<SysXxxVo>> list(SysXxxBo bo, PageQuery pageQuery) {
        return R.ok(xxxService.selectPageXxxList(bo, pageQuery));
    }

    @SaCheckPermission("system:xxx:query")
    @GetMapping(value = "/{xxxId}")
    public R<SysXxxVo> getInfo(@PathVariable Long xxxId) {
        return R.ok(xxxService.selectXxxById(xxxId));
    }

    @SaCheckPermission("system:xxx:add")
    @Log(title = "XXX管理", businessType = BusinessType.INSERT)
    @RepeatSubmit()
    @PostMapping
    public R<Void> add(@Validated @RequestBody SysXxxBo bo) {
        if (!xxxService.checkXxxNameUnique(bo))
            return R.fail("名称已存在");
        return toAjax(xxxService.insertXxx(bo));
    }

    @SaCheckPermission("system:xxx:edit")
    @Log(title = "XXX管理", businessType = BusinessType.UPDATE)
    @RepeatSubmit()
    @PutMapping
    public R<Void> edit(@Validated @RequestBody SysXxxBo bo) {
        if (!xxxService.checkXxxNameUnique(bo))
            return R.fail("名称已存在");
        return toAjax(xxxService.updateXxx(bo));
    }

    @SaCheckPermission("system:xxx:remove")
    @Log(title = "XXX管理", businessType = BusinessType.DELETE)
    @DeleteMapping("/{xxxIds}")
    public R<Void> remove(@PathVariable Long[] xxxIds) {
        return toAjax(xxxService.deleteXxxByIds(List.of(xxxIds)));
    }
}

8. QueryBuilder 常用模式

// 固定条件
.eq(SysUser::getDelFlag, SystemConstants.NORMAL)

// 可选条件（参数非空时才生效）
.eqIfPresent(SysUser::getUserId, bo.getUserId())      // 对象非null
.likeIfText(SysUser::getUserName, bo.getUserName())    // 字符串非空白
.eqIfText(SysUser::getStatus, bo.getStatus())          // 字符串非空白
.inIfNotEmpty(SysUser::getUserId, ids)                 // 集合非空

// 时间区间
.betweenParams(SysUser::getCreateTime, params, "beginTime", "endTime")

// 子条件组
.and(condition, w -> { w.eq(...).or().eq(...); })

// 排序
.orderByAsc(SysUser::getUserId)
