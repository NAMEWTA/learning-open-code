3.2 Service — 遵循标准模式

@Service
public class SysRoomServiceImpl implements ISysRoomService {

    // 分页查询 — 套用 QueryBuilder 模式
    public PageResult<SysRoomVo> selectPageRoomList(SysRoomBo room, PageQuery pageQuery) {
        Page<SysRoomVo> page = roomMapper.selectPageRoomList(pageQuery.build(), room);
        return PageResult.build(page.getRecords(), page.getTotal());
    }

    // 唯一性校验 — 套用标准模式
    public boolean checkRoomNameUnique(SysRoomBo room) {
        boolean exist = roomMapper.lambda()
            .eq(SysRoom::getRoomName, room.getRoomName())
            .eq(SysRoom::getDeptId, room.getDeptId())
            .neIfPresent(SysRoom::getRoomId, room.getRoomId())
            .exists();
        return !exist;
    }

    // 新增 — 套用事务 + 权限检查
    @Transactional(rollbackFor = Exception.class)
    public int insertRoom(SysRoomBo bo) {
        SysRoom room = MapstructUtils.convert(bo, SysRoom.class);
        return roomMapper.insert(room);
    }
}

3.3 Controller — 套用注解三件套

@Validated
@RequiredArgsConstructor
@RestController
@RequestMapping("/system/room")
public class SysRoomController extends BaseController {

    private final ISysRoomService roomService;
    private final ISysDeptService deptService;

    @SaCheckPermission("system:room:list")
    @GetMapping("/list")
    public R<PageResult<SysRoomVo>> list(SysRoomBo room, PageQuery pageQuery) {
        return R.ok(roomService.selectPageRoomList(room, pageQuery));
    }

    @SaCheckPermission("system:room:add")
    @Log(title = "会议室管理", businessType = BusinessType.INSERT)
    @RepeatSubmit()
    @PostMapping
    public R<Void> add(@Validated @RequestBody SysRoomBo room) {
        deptService.checkDeptDataScope(room.getDeptId());  // 部门权限检查
        if (!roomService.checkRoomNameUnique(room)) {
            return R.fail("新增会议室失败，会议室名称已存在");
        }
        return toAjax(roomService.insertRoom(room));
    }
}

// 在 Mapper 方法上加 @DataScope 注解（来自 common-mybatis）
public interface SysRoomMapper extends BaseMapperPlus<SysRoom, SysRoomVo> {

    @DataScope(deptAlias = "r")  // 🔑 自动为 r.dept_id 注入数据权限条件
    Page<SysRoomVo> selectPageRoomList(@Param("page") Page<SysRoom> page,
